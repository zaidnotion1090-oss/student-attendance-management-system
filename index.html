CONFIG 
import csv
import datetime
import os

FILE = "attendance.csv"
USERNAME = "admin"
PASSWORD = "1234"
LOGIN SYSTEM 
def login():
    print("=========== LOGIN REQUIRED ===========")
    for attempt in range(3):
        user = input("Enter username: ").strip()
        pwd = input("Enter password: ").strip()

        if user == USERNAME and pwd == PASSWORD:
            print("\nLogin successful!")
            return True
        else:
            print("Incorrect username or password.\n")
    return False
FILE HANDLING 
def create_file():
    if not os.path.exists(FILE):
        with open(FILE, "w", newline="") as f:
            writer = csv.writer(f)
            writer.writerow(["Date", "Student Name", "Status"])
MARK ATTENDANCE 
def mark_attendance():
    name = input("Enter student name: ").strip()
    status = input("Present (P) or Absent (A): ").upper().strip()

    today = datetime.date.today().strftime("%d-%m-%Y")

    with open(FILE, "a", newline="") as f:
        writer = csv.writer(f)
        writer.writerow([today, name, "Present" if status == "P" else "Absent"])
VIEW RAW TABLE 
def view_raw_table():
    with open(FILE, "r") as f:
        for row in csv.reader(f):
            print(row)
ATTENDANCE FREQUENCY 
def calculate_frequency():
    freq = {}
    with open(FILE, "r") as f:
        reader = csv.reader(f)
        next(reader)
        for date, name, status in reader:
            name_key = name.lower()
            if name_key not in freq:
                freq[name_key] = {"present": 0, "absent": 0}
            freq[name_key]["present" if status == "Present" else "absent"] += 1
    return freq
REPORT GENERATION 
def full_summary_report():
    freq = calculate_frequency()
    for name, data in freq.items():
        total = data["present"] + data["absent"]
        percent = (data["present"] / total * 100)
        print(name, percent)
MAIN MENU 
def menu():
    create_file()

    if not login():
        return

    while True:
        print("1. Mark Attendance")
        print("2. View Raw Attendance Table")
        print("3. Full Report")
        print("4. Exit")

        choice = input("Enter choice: ")

        if choice == "1": mark_attendance()
        elif choice == "2": view_raw_table()
        elif choice == "3": full_summary_report()
        elif choice == "4": break
